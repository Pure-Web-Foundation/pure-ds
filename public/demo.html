Toast API

// Get the pure-app element
const app = document.querySelector('pure-app');

// Show different toast types
app.toast('Message', 'information', 3000);
app.success('Success message', 3000);
app.warning('Warning message', 5000);
app.error('Error message', 5000);

// Or use legacy methods
app.toastSuccess('Success!');
app.toastInfo('Info!');
app.toastWarning('Warning!');
app.toastError('Error!');
