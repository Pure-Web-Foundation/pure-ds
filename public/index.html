Toast Notifications

Toast notifications appear in the top-right corner and auto-dismiss after a few seconds. Click the buttons below to see them in action:

Toast API

// Get the pure-app element
const app = document.querySelector('pure-app');

// Show toast notifications (duration is auto-calculated)
app.toast('Success message', { type: 'success' });
app.toast('Informational message', { type: 'info' });
app.toast('Warning message', { type: 'warning' });
app.toast('Error message', { type: 'error' });

// Options object (all optional):

 {
   type: 'info' | 'success' | 'warning' | 'error',  // default: 'info'
   duration: 3000,      // default: auto-calculated from message length
   closable: true,      // default: true
   persistent: false    // default: false (auto-dismiss)
 }
